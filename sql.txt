SELECT yr, subject, winner
  FROM nobel
 WHERE yr = 1950

SELECT winner
  FROM nobel
 WHERE yr = 1962
   AND subject = 'literature'

SELECT yr, subject
FROM nobel
WHERE winner ='Albert Einstein'

SELECT winner
FROM nobel
WHERE yr >= 2000
AND subject = 'Peace'

SELECT yr, subject, winner
FROM nobel
WHERE yr >=1980
AND yr <= 1989
AND subject = 'Literature'

SELECT * FROM nobel
 WHERE winner IN ('Theodore Roosevelt',
                  'Woodrow Wilson',
                  'Jimmy Carter',
                   'Barack Obama')

SELECT winner
FROM nobel
WHERE winner LIKE 'John%'


SELECT yr, subject, winner
FROM nobel
WHERE (subject = 'Physics' AND yr = 1980)
OR (subject ='Chemistry' AND yr = 1984)
